                        if (otherHeader !== header) {

                            const otherContent = otherHeader.nextElementSibling;

                            const otherIcon = otherHeader.querySelector('span:last-child');

                            otherContent.classList.remove('active');

                            otherIcon.classList.remove('rotate-45'); // Reset icon

                        }

                    });


                    // Toggle current one

                    content.classList.toggle('active');

                    icon.classList.toggle('rotate-45');

                });

            });

        });


        function copyToClipboard(text, element) {

            // Note: document.execCommand('copy') is used as navigator.clipboard might fail in some iFrame contexts.

            const textarea = document.createElement('textarea');

            textarea.value = text;

            textarea.style.position = 'fixed'; // Prevents scrolling to the bottom of the page

            document.body.appendChild(textarea);

            textarea.focus();

            textarea.select();

            try {

                const successful = document.execCommand('copy');

                if (successful) {

                    const originalText = element.innerHTML;

                    element.innerHTML = '¡Copiado!';

                    setTimeout(() => {

                        element.innerHTML = originalText;

                    }, 2000);

                } else {

                    console.error('Fallback: Copying text command was unsuccessful.');

                }

            } catch (err) {

                console.error('Fallback: Oops, unable to copy', err);

            }

            document.body.removeChild(textarea);

        }
